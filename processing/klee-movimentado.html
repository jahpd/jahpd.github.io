Klee Movimentado

Modifiquei algumas coisas no post anterior, para tornar o código um pouco mais complexo e bonito. A idéia é, ao invés de usar valores randômicos que estejam desconectados, os valores randômicos estarão conectados temporalmente.

Bezier_movimentados


function Mbezier(p){
	this.points = {x:{val:[],offset:[]},y:{val:[],offset:[]}}
	this.colors = {r:{val:0,offset:0},g:{val:0,offset:0},b:{val:0,offset:0},a:{val:0,offset:0}}
}

var n = 100;
var o = [];
var text;

function setup(){
	createCanvas(480, 360);
	for(var i=n-1;i>=0; i--){
		o[i] = new Mbezier(this);
		for(var j=3;j>=0; j--){
			o[i].points.x.offset[j] = Math.random();
			o[i].points.y.offset[j] = Math.random();
			o[i].points.x.val[j] = noise(o[i].points.x.offset[j]) * width;
			o[i].points.y.val[j] = noise(o[i].points.y.offset[j]) * height;
			o[i].colors.r.offset = Math.random();
			o[i].colors.g.offset = Math.random();
			o[i].colors.b.offset = Math.random();
			o[i].colors.a.offset = Math.random();
			o[i].colors.r.val = noise(o[i].colors.r.offset) * 255;
			o[i].colors.g.val = noise(o[i].colors.g.offset) * 255;
			o[i].colors.b.val = noise(o[i].colors.b.offset) * 255;
			o[i].colors.a.val = noise(o[i].colors.a.offset) * 255;
			
		}
	}
}

function draw(){
	background(0);
	for(var i=o.length-1;i>=0; i--){
		stroke(o[i].colors.r.val,
			o[i].colors.g.val,
			o[i].colors.b.val);
		noFill()
		bezier(o[i].points.x.val[0],
			o[i].points.y.val[0],
			o[i].points.x.val[1],
			o[i].points.y.val[1],
			o[i].points.x.val[2],
			o[i].points.y.val[2],
			o[i].points.x.val[3],
			o[i].points.y.val[3]);
		for(var j=3;j>=0; j--){
			o[i].points.x.offset[j] += 0.001
			o[i].points.y.offset[j] += 0.001
			o[i].points.x.val[j] = noise(o[i].points.x.offset[j]) * width;
			o[i].points.y.val[j] = noise(o[i].points.y.offset[j]) * height;
			o[i].colors.r.offset += 0.001;
			o[i].colors.g.offset += 0.001;
			o[i].colors.b.offset += 0.001;
			o[i].colors.a.offset += 0.001;
			o[i].colors.r.val = noise(o[i].colors.r.offset) * 255;
			o[i].colors.g.val = noise(o[i].colors.g.offset) * 255;
			o[i].colors.b.val = noise(o[i].colors.b.offset) * 255;
			o[i].colors.a.val = noise(o[i].colors.a.offset) * 255;
		}
	}
}

Isto é, o que aparece em um dado frame, depende do frame anterior. Para isso usei a função noise. Esta função gera um Perlin Noise, ou um ruído procedural.

O efeito disso é uma imagem que se movimenta aos poucos, e cujas cores mudam gradualmente.
