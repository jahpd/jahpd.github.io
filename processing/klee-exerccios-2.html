function _(x){return random(x);}
function rx(){return _(width);}
function ry(){return _(height);}
function rc(){return _(255);}
function setup(){
  createCanvas(480, 360);
  frameRate(0.5);
}
function draw(){ 
  background(0);
  var i, r,g,b,x,y;
  var _r,_g,_b,_x,_y;
  for(i=0; i < 20; i++) {
    r = rc(); g = rc(); b = rc();
    x = rx(); y = ry();
    stroke(max(_r||0,r),max(_g||0,g),max(_b||0,b), rc());
    fill(0, 0, 0, rc());
    bezier(x,y,rx(),ry(),rx(),ry(),_x||width/2,_y||height/2);
    _x = x; _y = y; _r = r; _g = g; _b = b;
  }
}
